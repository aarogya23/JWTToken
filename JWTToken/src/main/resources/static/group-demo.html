📋 How Groups Are Loaded:

// JavaScript in groups.html
function loadGroups() {
    fetch('/api/groups', {
        headers: {
            'Authorization': 'Bearer ' + localStorage.getItem('jwtToken')
        }
    })
    .then(response => response.json())
    .then(groups => {
        const groupsDiv = document.getElementById('groups');
        groupsDiv.innerHTML = '';

        if (groups.length === 0) {
            groupsDiv.innerHTML = '<p>No groups yet. Create one above!</p>';
        } else {
            groups.forEach(group => {
                const div = document.createElement('div');
                div.className = 'group-item';
                div.innerHTML = `<strong>${group.name}</strong><small>Created by: ${group.createdBy.fullName}</small>`;
                div.onclick = () => joinGroup(group.id);
                groupsDiv.appendChild(div);
            });
        }
    })
    .catch(error => {
        console.error('Error loading groups:', error);
        alert('Error loading groups');
    });
}
